# 寻找两个正序数组的中位数

给定两个大小分别为 `m` 和 `n` 的正序（从小到大）数组 `nums1` 和 `nums2`。请你找出并返回这两个正序数组的 **中位数** 。

```
import java.util.*;

public class Solution {
    public double findMedianSortedArrays(int[] nums1, int[] nums2) {

        if(nums1.length == 0 && nums2.length == 1) return nums2[0];
        if(nums2.length == 0 && nums1.length == 1) return nums1[0];
        if(nums1.length == 0 && nums2.length == 0) return 0;


        //先将int数组转换为数值流
        IntStream stream1 = Arrays.stream(nums1);
        //流中的元素全部装箱，转换为流 ---->int转为Integer
        Stream<Integer> integerStream1 = stream1.boxed();
        //将流转换为数组
        Integer[] integers1 = integerStream1.toArray(Integer[]::new);

        //先将int数组转换为数值流
        IntStream stream2 = Arrays.stream(nums2);
        //流中的元素全部装箱，转换为流 ---->int转为Integer
        Stream<Integer> integerStream2 = stream2.boxed();
        //将流转换为数组
        Integer[] integers2 = integerStream2.toArray(Integer[]::new);

        ArrayList<Integer> nums = new ArrayList<>();

        if(integers1.length > 0)
            nums.addAll(Arrays.asList(integers1));
        if(integers2.length > 0)
            nums.addAll(Arrays.asList(integers2));

        Collections.sort(nums);

    }
}
```

* 通过指针进行判断

```

class Solution {
       public double findMedianSortedArrays(int[] nums1, int[] nums2) {

        int n = nums1.length;
        int m = nums2.length;

        int a,b;
        a = (int)(n+m+1) / 2 - 1;
        b = (n+m)%2==0?a+1:a;

        int ans = 0;
        int count = 0;

        int i,j,k;
        i=j=k=0;

        while (i<n&&j<m){
            if(nums1[i] < nums2[j]){
                if(k == a || k == b){
                    ans += nums1[i];
                    count++;
                }
                i++;k++;
            }
            else {
                if(k == a || k == b){
                    ans += nums2[j];
                    count++;
                }
                j++;k++;
            }

        }

        while(i<n){
            if(k == a || k == b){
                ans += nums1[i];
                count++;
            }
            i++;k++;
        }

        while(j<m){
            if(k == a || k == b){
                ans += nums2[j];
                count++;
            }
            j++;k++;
        }

        return ans==0?ans:(double) ans/count;
    }
}
```
