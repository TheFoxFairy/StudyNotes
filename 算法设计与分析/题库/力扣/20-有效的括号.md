class Solution {
    public boolean select(char c1,char c2){
        if(c2 == '}' && c1 == '{') return true;
        if(c2 == ']' && c1 == '[') return true;
        if(c2 == ')' && c1 == '(') return true;
        return false;
    }

    public boolean isValid(String s) {

        Stack<Character> stack = new Stack<>();

        boolean flag = false;
        for(int i=0;i<s.length();i++){
            char c2 = s.charAt(i);

            if(stack.empty()) stack.push(c2);
            else{
                char c1 = stack.peek();
                if(select(c1,c2)){
//                    System.out.println(c1+":"+c2+":"+true);
                    stack.pop();
                }
                else{
                    stack.push(c2);
                }
            }
        }
//        while(!stack.empty()){
//            System.out.println(stack.peek());
//            stack.pop();
//        }
        if(stack.size() > 0) return false;
        else return true;
    }
}
