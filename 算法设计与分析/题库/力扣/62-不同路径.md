# [不同路径](https://leetcode-cn.com/problems/unique-paths/)

![image-20210609134240630](assets/Untitled/image-20210609134240630.png)

```
class Solution {
    public int uniquePaths(int m, int n) {
        int result = 0;
        int[][] df = new int[m][n];
        for (int i = 0; i < m; i++)
            for (int j = 0; j < n; j++)
                if (i == 0 || j == 0)
                    df[i][j] = 1;
                else
                    df[i][j] = df[i - 1][j] + df[i][j - 1];
        // System.out.println(df[m-1][n-1]);
        return df[m-1][n-1];
    }
}
```
