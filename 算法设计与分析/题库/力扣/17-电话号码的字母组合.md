class Solution{
    public List<String> addLetter(String data, List<String> list) {
        List<String> new_list = new ArrayList<>();
        if (list.size() == 0) {
            for (int i = 0; i < data.length(); i++)
                new_list.add(data.charAt(i) + "");
        } else {
            int n = list.size();
            for (int j = 0; j < n; j++) {
                    String sc = list.get(j);
                for (int i = 0; i < data.length(); i++) {
                    new_list.add(sc + data.charAt(i));
                }

            }
        }
        return new_list;
    }

    public List<String> letterCombinations(String digits) {

        if (digits.length() > 4 || digits.length() < 0) return null;

        String[] s = {"abc", "def", "ghi", "jkl", "mno", "pqrs", "tuv", "wxyz"};
        List<String> list = new ArrayList<>();
        for (int i = 0; i < digits.length(); i++) {
            String sc = s[digits.charAt(i) - '0' - 2];
            list = addLetter(sc, list);
        }
//        for (String sc : list) {
//            System.out.println(sc);
//        }
        return list;
    }
}
