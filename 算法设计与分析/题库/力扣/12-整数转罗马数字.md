class Solution {
    public String selectRoman(String num) {
//        System.out.println(num);

        int current_num = Integer.parseInt(num);

        if (current_num == 0) return "";

        int first_num = Integer.parseInt(num.charAt(0)+"");

        if (current_num < 5) {
            if (current_num == 4) return "IV";

            String s = "";
            for (int i = 0; i < current_num; i++)
                s += "I";
            return s;
        } else if (current_num < 10) {
            if (current_num == 9) return "IX";
            return "V" + selectRoman(current_num - 5 + "");
        } else if (current_num < 50) {
            if (first_num == 4) return "XL"+selectRoman((current_num-40)+"");
            String s = "";
            for (int i = 0; i < Math.ceil(current_num / 10); i++)
                s += "X";
            return s+selectRoman((int)(current_num - Math.ceil(current_num / 10)*10) + "");

        } else if (current_num < 100) {
            if (first_num == 9) return "XC"+selectRoman((current_num-90)+"");
            return "L" + selectRoman(current_num - 50 + "");
        }
        else if (current_num < 500) {
            if (first_num == 4) return "CD"+selectRoman((current_num-400)+"");
            String s = "";
            for (int i = 0; i < Math.ceil(current_num / 100); i++)
                s += "C";
            return s+selectRoman((int)(current_num - Math.ceil(current_num / 100)*100) + "");

        } else if (current_num < 1000) {
            if (first_num == 9) return "CM"+selectRoman((current_num-900)+"");
            return "D" + selectRoman(current_num - 500 + "");
        }else {
            int number = (int)current_num / 1000;
            String s = "";
            for(int i=0;i<number;i++)
                s += "M";
            return s+selectRoman((current_num-number*1000)+"");
        }
    }

    public String intToRoman(int num) {
        if (num < 1 || num > 3999) return "";
        String s = selectRoman(num + "");
//        System.out.println(s);
        return s;
    }
}
