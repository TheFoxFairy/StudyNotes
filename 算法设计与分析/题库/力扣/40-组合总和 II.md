class Solution {
       List<List<Integer>> result = new ArrayList<>();

    public void total(int[] candidates, int start, int target, List<Integer> list) {

        if (target == 0) {
            result.add(new ArrayList<>(list));
        } else {
            for (int i = start; i < candidates.length && target - candidates[i] >=0;) {
                list.add(candidates[i]);
                total(candidates, i + 1, target - candidates[i], list);
                list.remove(list.size() - 1);
                i++;
                while (i<candidates.length && candidates[i-1] == candidates[i]) i++;
            }
        }
    }

    public List<List<Integer>> combinationSum2(int[] candidates, int target) {
        int len = candidates.length;
        List<Integer> list = new ArrayList<>();

        Arrays.sort(candidates);

        total(candidates, 0, target, list);

        return result;
    }
}
