import java.util.PriorityQueue;
class Solution {
    class ListNodeStatus implements Comparable<ListNodeStatus>{
        ListNode node;
        int val;
        ListNodeStatus(ListNode node,int val){
            this.node = node;
            this.val = val;
        }

        public int compareTo(ListNodeStatus lns){
            return this.val - lns.val;
        }
    }

    public ListNode mergeKLists(ListNode[] lists) {
        Queue<ListNodeStatus> queue = new PriorityQueue<>();
        ListNode head,tail;
        head = tail = new ListNode();

        for(ListNode node:lists)
            if(node != null)
                queue.add(new ListNodeStatus(node,node.val));

        while(!queue.isEmpty()){

            ListNodeStatus status = queue.peek();
            ListNode next = status.node.next;
            
            ListNode node = new ListNode(status.val);
            tail.next = node;
            tail = node;

            // System.out.println(status.val);
            queue.remove();
            if(next != null)
                queue.add(new ListNodeStatus(next,next.val));

            
        }
        head = head.next;
        return head;
    }
}
