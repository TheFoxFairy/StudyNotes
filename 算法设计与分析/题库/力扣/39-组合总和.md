# [组合总和](https://leetcode-cn.com/problems/combination-sum/)

class Solution {
    List<List<Integer>> result = new ArrayList<>();

    public int total(int[] candidates, int start, int target, List<Integer> list) {

        if (start < candidates.length && target == 0) {
            result.add(new ArrayList<>(list));
            return 1;
        }

        if (start == candidates.length || target < 0)
            return 0;

        list.add(candidates[start]);
        total(candidates, start, target - candidates[start], list);
        target += candidates[start];
        list.remove(list.size() - 1);

        return total(candidates, start + 1, target - candidates[start], list);
    }

    public List<List<Integer>> combinationSum(int[] candidates, int target) {
        int len = candidates.length;
        List<Integer> l = new ArrayList<>();

        Arrays.sort(candidates);

        for (int i = 0; i < len; i++) {
            l.add(candidates[i]);
            total(candidates, i, target - candidates[i], l);
            l.remove(l.size() - 1);
        }

//        for(List<Integer> r:result){

//                System.out.println(r.get(i));
//            }
//            System.out.println();
//        }
        return result;
    }
}
