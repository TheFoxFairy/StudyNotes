# [跳跃游戏](https://leetcode-cn.com/problems/jump-game/)

![image-20210605171533348](assets/55-%E8%B7%B3%E8%B7%83%E6%B8%B8%E6%88%8F/image-20210605171533348.png)

```
class Solution {
    public boolean canJump(int[] nums) {
        int step = nums[0];
        int i = 1;
        while (i < nums.length && step > 0){
            step--;
            if(step == 0 && nums[i] == 0){i++;break;}
            step = Math.max(nums[i],step);
            i++;
        }
        boolean flag = (i == nums.length)?true:false;
        return flag;
    }
}
```
