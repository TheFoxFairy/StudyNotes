# 无重复字符的最长子串

给定一个字符串，请你找出其中不含有重复字符的 **最长子串** 的长度。

![image-20210414155338530](assets/03-%E6%97%A0%E9%87%8D%E5%A4%8D%E5%AD%97%E7%AC%A6%E7%9A%84%E6%9C%80%E9%95%BF%E5%AD%90%E4%B8%B2/image-20210414155338530.png)

```
import java.util.HashSet;

public class Solution {
    public int lengthOfLongestSubstring(String s) {

        if(s.length() == 0) return 0;
        else if(s.length() == 1) return 1;

        int n = s.length();
        int ans = 0;
        int k = -1;

        HashSet<Character> set = new HashSet<>();

        for(int i=0;i<n;i++){

            if(i!=0){
                set.remove(s.charAt(i-1));
            }

            while(k+1 < n && !set.contains(s.charAt(k+1))){
                set.add(s.charAt(k+1));
                k++;
            }

            ans = Math.max(k - i + 1,ans);

        }
//        System.out.println(ans);
        return ans;
    }
}
```
