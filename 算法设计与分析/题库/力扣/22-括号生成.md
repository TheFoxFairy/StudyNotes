    public void generateSubParenthesis(List<String> ans,StringBuffer cur,int open,int close, int max){
        if(cur.length() == max * 2){
            ans.add(cur.toString());
            return;
        }
        // 先插入open的
        if(open < max){
            cur.append('(');
            generateSubParenthesis(ans,cur,open+1,close,max);
            cur.deleteCharAt(cur.length()-1);//回溯
        }
        //在插入close的
        if(close < open){
            cur.append(')');
            generateSubParenthesis(ans,cur,open,close+1,max);
            cur.deleteCharAt(cur.length()-1);//回溯
        }
    }


    public List<String> generateParenthesis(int n) {
        List<String> ans = new ArrayList<>();
        generateSubParenthesis(ans,new StringBuffer(),0,0,n);
//        for(String s:ans)
//            System.out.println(s);
        return ans;
    }
