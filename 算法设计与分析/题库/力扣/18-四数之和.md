# [四数之和](https://leetcode-cn.com/problems/4sum/)

class Solution {
    public List<List<Integer>> fourSum(int[] nums, int target) {
        if(nums.length < 0||nums.length > 200) return null;

        Arrays.sort(nums);

        int n = nums.length;

        List<List<Integer>> list = new ArrayList<>();

        for(int first=0;first<n;first++){
            if(first > 0 && nums[first] == nums[first-1]) continue;

            for(int second=first+1;second<n;second++){
                if(second > first+1 && nums[second] == nums[second-1]) continue;

                int fourth = n-1;

                for(int third=second+1;third<n;third++){

                    if(third > second+1 && nums[third] == nums[third-1]) continue;

                    while(third < fourth && nums[first]+nums[second]+nums[third]+nums[fourth] > target) fourth--;

                    if(fourth == third) break;

                    if(target == (nums[first]+nums[second]+nums[third]+nums[fourth])) {
                        List<Integer> l = new ArrayList<>();
                        l.add(nums[first]);
                        l.add(nums[second]);
                        l.add(nums[third]);
                        l.add(nums[fourth]);
                        list.add(l);
//                        System.out.println(nums[first] + ":" + nums[second] + ":" + nums[third] + ":" + nums[fourth]);
                    }
                }
            }
        }
//        for(List<Integer>l:list){
//            for(Integer num:l){
//                System.out.print(num+" ");
//            }
//            System.out.println();
//        }
        return list;
    }
}
