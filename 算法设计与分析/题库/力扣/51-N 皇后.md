class Solution {
 public void Queens(int i, int n, char[][] result, int[] labels,List<List<String>> ans) {

        if (i == n) {
            List<String> list = new ArrayList<>();
            for (int k = 0; k < n; k++) {
                String s = String.valueOf(result[k]);
                list.add(s);
//                System.out.println(s);
            }
            ans.add(list);
        }


            for (int j = 0; j < n; j++) {
            /*
            前n个是用来确定行和列的
            后n个到4n个是用来扫描正向斜的
            后4n个到5n是用来扫描反向斜的
            */
                if(labels[j] == 0 && labels[i+j+n] == 0 && labels[4*n+i-j] == 0){
                    labels[j] = 1;
                    labels[i+j+n] = 1;
                    labels[4*n+i-j] = 1;
                    result[i][j] = 'Q';
                    Queens(i+1,n,result,labels,ans);
                    result[i][j] = '.';
                    labels[4*n+i-j] = 0;
                    labels[i+j+n] = 0;
                    labels[j] = 0;
                }

        }
    }

    public List<List<String>> solveNQueens(int n) {

        char[][] result = new char[n][n];

        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                result[i][j] = '.';
            }
        }
        int[] labels = new int[5*n];
        List<List<String>> ans = new ArrayList<List<String>>();
        Queens(0, n, result, labels,ans);
        return ans;
    }
}
