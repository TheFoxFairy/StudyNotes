# 最长回文子串

给你一个字符串 `s`，找到 `s` 中最长的回文子串。

![image-20210415133054773](assets/05-%E6%9C%80%E9%95%BF%E5%9B%9E%E6%96%87%E5%AD%90%E4%B8%B2/image-20210415133054773.png)

```
public class Solution {

    public Integer Palindrome(int start,String s){
        int i = start;
        int j = s.length() - 1;
        int total = 1;
        int end = j;
        while(i<j) {
//            System.out.println(s.charAt(i)+":"+s.charAt(j));
            if(s.charAt(i) != s.charAt(j)) {
                i=start;
                total = 1;
                j=end-1;
                end--;
            }
            else{
                i++;
                total += 2;
                j--;
            }
        }

        total = i>j?total-1:total;
//        System.out.println(start+":"+total);
        return total;
    }

    public String longestPalindrome(String s) {
        int n = s.length();
        int i = 0;
        int min = 0;
        int index = -1;
        while (i<n){
            Integer total = Palindrome(i,s);
            if(min < total){
                min = total;
                index = i;
            }
            i++;
        }
//        System.out.println(s.substring(index,index+min));
        return s.substring(index,index+min);
    }
}
```
