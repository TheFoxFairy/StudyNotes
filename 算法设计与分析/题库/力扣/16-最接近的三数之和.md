# [ 最接近的三数之和](https://leetcode-cn.com/problems/3sum-closest/)

![image-20210419230231132](assets/16-%E6%9C%80%E6%8E%A5%E8%BF%91%E7%9A%84%E4%B8%89%E6%95%B0%E4%B9%8B%E5%92%8C/image-20210419230231132.png)

```
class Solution {
    public int threeSumClosest(int[] nums, int target) {
        int n = nums.length;
        int minNum = Integer.MAX_VALUE;
        int ans = 0;

        for(int first=0;first < n;first++){
            for(int second=first+1;second<n;second++){
                for(int third=second+1;third<n;third++) {
                    int total = Math.abs(target - (nums[first] + nums[second] + nums[third]));
                    if (total < minNum) {
                        minNum = total;
                        ans = nums[first] + nums[second] + nums[third];
                    }
                    if (total == 0) break;
                }
                if (minNum == 0) break;
            }
        }
//        System.out.println(ans);
        return ans;
    }
}
```
