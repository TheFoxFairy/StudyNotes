# [合并区间](https://leetcode-cn.com/problems/merge-intervals/)

class Solution {
   public boolean check(int i,int j,int[][] intervals){
//        System.out.println(intervals[j][0]+":"+intervals[j][1]+":"+intervals[i][0]+":"+intervals[i][1]);
        if(intervals[j][1] < intervals[i][0] || intervals[j][0] > intervals[i][1]) {
            return true;
        }
        else{
            return false;
        }
    }

    public int[][] merge(int[][] intervals) {
        int total = 1;
        //行排序
        Arrays.sort(intervals, new Comparator<int[]>() {
            @Override
            public int compare(int[] o1, int[] o2) {
                if (o1[0]==o2[0]) return o1[1]-o2[1];
                return o1[0]-o2[0];
            }
        });
        for(int i=1;i<intervals.length;i++){
            int flag = 1;
            for(int j=0;j<total;j++){
                if(check(i,j,intervals)){
                    continue;
                }else{
                    int newi = Math.min(intervals[i][0],intervals[j][0]);
                    int newj = Math.max(intervals[i][1],intervals[j][1]);
                    intervals[j][0] = newi;
                    intervals[j][1] = newj;
                    flag = 0;
                    break;
                }
            }
            if(flag == 1){
                intervals[total][0] = intervals[i][0];
                intervals[total][1] = intervals[i][1];
                total++;
            }
        }
        int[][] result = new int[total][2];
        for(int i=0;i<total;i++){
            result[i][0] = intervals[i][0];
            result[i][1] = intervals[i][1];
        }
        return result;
    }
}
