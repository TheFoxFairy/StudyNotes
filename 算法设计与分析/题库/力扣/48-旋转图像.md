class Solution {
    public void swap(int ra,int ca,int rb,int cb,int[][] matrix){
        int tmp = matrix[ra][ca];
        matrix[ra][ca] = matrix[rb][cb];
        matrix[rb][cb] = tmp;
    }

    public void rotate(int[][] matrix) {
        int n = matrix.length;

        for(int i=0;i<Math.ceil(n/2);i++){
            for(int j=0;j<n;j++){
                swap(i,j,n - i - 1,j,matrix);
            }
        }

        for(int i=0;i<n;i++){
            for(int j=0;j<i;j++){
                swap(i,j,j,i,matrix);
            }
        }
//        for(int i=0;i<n;i++){

//            }
//            System.out.println();
//        }
    }
}
